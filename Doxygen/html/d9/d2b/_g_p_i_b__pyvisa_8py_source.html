27def psend(txt_cmd):
28 print("Psend input:", txt_cmd)
29 sum = 0 # variable for check summ
30 cmd = [] # array for formatted command
31
32 for item in txt_cmd:
33 # sum = sum + hex(item.encode('utf-8').hex())
34 #print(f"Symbol: {item}, value: {ord(item)}")
35 sum = sum + int(ord(item))
36 cmd.append(ord(item))
37 check_sum = sum & 0x00FF
38 print(f"sum: {sum}, check_sum: {check_sum}, cmd: {cmd}")
39 # If the checksum is less or equal to 0x20, 0x20 is added again.
40 # Thus ensures that the checksum is not interpreted as control character.
41 if check_sum <= 0x20:
42 check_sum += 0x20
43 # additional protocol requirements
44 # STX=0x02 + Command + ETX=0x03 + CheckSum
45 cmd.insert(0, 2) # insertion of STX=0x02 to a first place
46 cmd.append(3) # termination message with ETX=0x03
47 cmd.append(check_sum) # adding check sum at the end
48 print(f"protocol cmd: {cmd}")
49 print(bytearray(cmd))
50 return (bytes(cmd))
