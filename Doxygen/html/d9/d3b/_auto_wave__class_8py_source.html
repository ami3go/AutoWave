18def range_check(val, min, max, val_name):
19 """
20 Checking if the value inside the range min>=val<=max
21
22 :param val: input value
23 :param min: minimal value to check with
24 :param max: maxinum value to check with
25 :param val_name: name of parameter. In case of out of range event the name will be printer to help find an error
26 :type val_name: string
27 :return: return Val if in the range, val=max if >=max, val=min if val<=min
28 """
29 if val > max:
30 print(f"Wrong {val_name}: {val}. Max value should be less then {max}")
31 val = max
32 if val < min:
33 print(f"Wrong {val_name}: {val}. Should be >= {min}")
34 val = min

49def dec_array2check_sum(dec_array):
50 """
51 Calculate a check sum for decimal array
52
53 :param dec_array: binary array
54 :return: calclulated check summ according to documentation
55 """
56 check_sum = sum(dec_array) & 0x00FF
57 if check_sum <= 0x20:
58 check_sum += 0x20
59 return check_sum

148 def psend(self, txt_cmd):
149 """
150 Sending protocol command: STX=0x02 + Command + ETX=0x03 + CheckSum
151 :param txt_cmd: VISA string command
152 :type txt_cmd: str
153 :return: None
154 """
155 cmd = str2dec_array(txt_cmd)
156 cmd.insert(0, 2) # insertion of STX=0x02 to a first place
157 cmd.append(3) # termination message with ETX=0x03
158 cmd.append(str2check_sum(txt_cmd)) # adding check sum at the end
159 # print(f"protocol cmd: {cmd}")
160 # print(bytes(cmd))
161 self.instinst.write_raw(bytes(cmd))
162 delay()

164 def pquery(self, cmd_str, err_check=False, p_check=True):
165 '''
166 Delay and retry in cause of old device with slow processing time
167 In case of error a 10 attempts will be made before everything will get crashed.
168
169 :param cmd_str: VISA string command
170 :type cmd_str: str
171 :return: instument replay string
172 :rtype: str
173 '''
174 for i in range(10):
175 try:
176 # debug print to check how may tries
177 # print("trying",i)
178 self.psendpsend(cmd_str)
179 delay() # regular delay according to datasheet before next command
180 return_raw = None
181 return_raw = self.instinst.read_raw()
182 # print("read_raw:",return_raw )
183 # check return line
184 if len(return_raw) == 1:
185 if return_raw == b'\x19':
186 raise Exception("Device: BUSY")
187 if return_raw == b'\x16':
188 raise Exception("Device: NOT READY")
189 if return_raw == b'\x15':
190 raise Exception("Device: Negative ACKnowledge")
191 if p_check is True:
192 # TBD: require to add check sum check
193 check_sum = dec_array2check_sum(return_raw[1:-2])
194 if (return_raw[0] == 0x02) and (return_raw[-2] == 0x03) and (check_sum == return_raw[-1]):
195 # # typical val = b'\x02TRIG:GEN 1\x03\x9b'
196 # # x02 - start symbol
197 # # x03 - stop symbol
198 # # x9b - check sum
199 # print("p_check:", return_raw)
200 return_raw = return_raw[1:-2]
201 return_str = return_raw.decode("utf-8")
202 if err_check is True:
203 if (return_str.find(":ERR") != -1):
204 raise Exception("Error in replay")
205 return return_str
206
207 else:
208 raise Exception("Error in start(0x02)/end(0x03) terminator")
209
210
211 return return_raw.decode("utf-8")
212
213 except Exception as e:
214 print(f"Pquery[{i}]: {cmd_str}, Reply: {return_raw}, Error: {e}")
215 delay(5)
